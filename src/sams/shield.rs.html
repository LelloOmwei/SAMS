1//! SAMS Shield - Data anonymization and PQC protection

3//! This module provides semantic obfuscation, anonymization, and post-quantum
4//! cryptography features extracted and enhanced from the OMWEI-MVP project.

9/// Error types for shield operations
10#[derive(Debug, Clone, PartialEq)]
11pub enum ShieldError {
12    /// Invalid key size
13    InvalidKeySize,
14    /// Invalid signature
15    InvalidSignature,
16    /// Cryptographic operation failed
17    CryptoError,
18    /// Buffer size mismatch
19    BufferSizeError,
20}
21
22impl core::fmt::Display for ShieldError {
23    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
24        match self {
25            ShieldError::InvalidKeySize => write!(f, "Invalid key size"),
26            ShieldError::InvalidSignature => write!(f, "Invalid signature"),
27            ShieldError::CryptoError => write!(f, "Cryptographic operation failed"),
28            ShieldError::BufferSizeError => write!(f, "Buffer size mismatch"),
29        }
30    }
31}

36/// Anonymization levels for semantic atoms
37#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
38pub enum AnonymizationLevel {
39    /// No anonymization - raw data
40    None = 0,
41    /// Basic anonymization - entity ID masking
42    Basic = 1,
43    /// Medium anonymization - entity + node ID masking
44    Medium = 2,
45    /// High anonymization - all identifying information masked
46    High = 3,
47    /// Maximum anonymization - complete data obfuscation
48    Maximum = 4,
49}
50
51/// Semantic obfuscation strategies
52#[derive(Debug, Clone, Copy)]
53pub enum ObfuscationStrategy {
54    /// Replace with zeros
55    Zeroize,
56    /// Add random noise
57    Noise,
58    /// Use hash-based masking
59    HashMask,
60    /// Apply differential privacy
61    DifferentialPrivacy,
62    /// Round to nearest bucket
63    Bucketize,
64}
65
66/// Shield configuration for atom protection
67#[derive(Debug, Clone)]
68pub struct ShieldConfig {
69    /// Anonymization level
70    pub anonymization_level: AnonymizationLevel,
71    /// Obfuscation strategy
72    pub obfuscation_strategy: ObfuscationStrategy,
73    /// Enable PQC signing
74    pub enable_pqc: bool,
75    /// Privacy budget for differential privacy
76    pub privacy_budget: f64,
77}
78
79impl Default for ShieldConfig {
80    fn default() -> Self {
81        Self {
82            anonymization_level: AnonymizationLevel::Medium,
83            obfuscation_strategy: ObfuscationStrategy::HashMask,
84            enable_pqc: true,
85            privacy_budget: 1.0,
86        }
87    }

90/// SAMS Shield - Main protection interface
91pub struct Shield {
92    config: ShieldConfig,
93    #[cfg(feature = "pqc")]
94    hmac_key: [u8; 32],
95}
96
97impl Shield {
98    /// Create a new shield with default configuration
99    pub fn new() -> Self {
100        Self::with_config(ShieldConfig::default())
101    }
102
103    /// Create a new shield with custom configuration
104    pub fn with_config(config: ShieldConfig) -> Self {
105        Self {
106            config,
107            #[cfg(feature = "pqc")]
108            hmac_key: [0u8; 32], // Will be initialized in set_key
109        }
110    }
111
112    /// Set the HMAC key for PQC operations
113    #[cfg(feature = "pqc")]
114    pub fn set_key(&mut self, key: &[u8]) -> Result<()> {
115        if key.len() != 32 {
116            return Err("Invalid key size".into());
117        }
118        self.hmac_key.copy_from_slice(key);
119        Ok(())
120    }
121
122    /// Anonymize a semantic atom
123    pub fn anonymize_atom(&self, atom: &mut SemanticAtom) -> Result<()> {
124        match self.config.anonymization_level {
125            AnonymizationLevel::None => {
126                // No anonymization
127            }
128            AnonymizationLevel::Basic => {
129                self.anonymize_basic(atom)?;
130            }
131            AnonymizationLevel::Medium => {
132                self.anonymize_basic(atom)?;
133                self.anonymize_medium(atom)?;
134            }
135            AnonymizationLevel::High => {
136                self.anonymize_basic(atom)?;
137                self.anonymize_medium(atom)?;
138                self.anonymize_high(atom)?;
139            }
140            AnonymizationLevel::Maximum => {
141                self.anonymize_maximum(atom)?;
142            }
143        }
144        Ok(())
145    }
146
147    /// Apply semantic obfuscation to atom values
148    pub fn obfuscate_values(&self, atom: &mut SemanticAtom) -> Result<()> {
149        match self.config.obfuscation_strategy {
150            ObfuscationStrategy::Zeroize => {
151                atom.value_fixed = 0;
152            }
153            ObfuscationStrategy::Noise => {
154                self.add_noise(atom)?;
155            }
156            ObfuscationStrategy::HashMask => {
157                self.hash_mask_value(atom)?;
158            }
159            ObfuscationStrategy::DifferentialPrivacy => {
160                self.apply_differential_privacy(atom)?;
161            }
162            ObfuscationStrategy::Bucketize => {
163                self.bucketize_value(atom)?;
164            }
165        }
166        Ok(())
167    }
168
169    /// Sign an atom with PQC (HMAC-SHA256 for now, will be ML-DSA)
170    #[cfg(feature = "pqc")]
171    pub fn sign_atom(&self, atom: &SemanticAtom) -> Result<[u8; 32]> {
172        use sha2::Sha256;
173        use hmac::{Hmac, Mac};
174
175        let mut mac = Hmac::<Sha256>::new_from_slice(&self.hmac_key)
176            .map_err(|_| "Crypto error")?;
177        mac.update(atom.as_bytes());
178        let result = mac.finalize();
179        let bytes = result.into_bytes();
180        
181        let mut signature = [0u8; 32];
182        signature.copy_from_slice(&bytes);
183        Ok(signature)
184    }
185
186    /// Verify an atom's PQC signature
187    #[cfg(feature = "pqc")]
188    pub fn verify_atom(&self, atom: &SemanticAtom, signature: &[u8]) -> Result<bool> {
189        if signature.len() != 32 {
190            return Err("Invalid signature".into());
191        }
192
193        let expected_signature = self.sign_atom(atom)?;
194        Ok(expected_signature.as_slice() == signature)
195    }
196
197    /// Create a protected atom packet (atom + signature)
198    #[cfg(feature = "pqc")]
199    pub fn create_protected_packet(&self, atom: &SemanticAtom) -> Result<[u8; crate::ATOM_SIZE + 32]> {
200        let signature = self.sign_atom(atom)?;
201        let mut packet = [0u8; crate::ATOM_SIZE + 32];
202        
203        // Copy atom data
204        packet[..crate::ATOM_SIZE].copy_from_slice(atom.as_bytes());
205        // Copy signature
206        packet[crate::ATOM_SIZE..].copy_from_slice(&signature);
207        
208        Ok(packet)
209    }
210
211    /// Extract and verify atom from protected packet
212    #[cfg(feature = "pqc")]
213    pub fn extract_protected_packet(&self, packet: &[u8]) -> Result<(SemanticAtom, bool)> {
214        if packet.len() != crate::ATOM_SIZE + 32 {
215            return Err("Buffer size error".into());
216        }
217
218        let atom_data = &packet[..crate::ATOM_SIZE];
219        let signature = &packet[crate::ATOM_SIZE..];
220
221        let atom = SemanticAtom::from_bytes(atom_data)?;
222        let is_valid = self.verify_atom(&atom, signature)?;
223
224        Ok((atom.clone(), is_valid))
225    }
226
227    // Private anonymization methods
228
229    fn anonymize_basic(&self, atom: &mut SemanticAtom) -> Result<()> {
230        // Mask entity ID with hash
231        atom.entity_id = self.hash_u32(atom.entity_id);
232        Ok(())
233    }
234
235    fn anonymize_medium(&self, atom: &mut SemanticAtom) -> Result<()> {
236        // Mask node ID
237        atom.node_id = self.hash_u16(atom.node_id);
238        Ok(())
239    }
240
241    fn anonymize_high(&self, atom: &mut SemanticAtom) -> Result<()> {
242        // Mask sequence number
243        atom.sequence = self.hash_u32(atom.sequence);
244        // Coarsen timestamp to nearest hour
245        const HOUR_US: u64 = 3_600_000_000; // 1 hour in microseconds
246        atom.timestamp_us = (atom.timestamp_us / HOUR_US) * HOUR_US;
247        Ok(())
248    }
249
250    fn anonymize_maximum(&self, atom: &mut SemanticAtom) -> Result<()> {
251        // Complete obfuscation - keep only telemetry type and basic structure
252        atom.entity_id = 0;
253        atom.sequence = 0;
254        atom.value_fixed = 0;
255        atom.status_flags = 0;
256        atom.timestamp_us = 0;
257        atom.node_id = 0;
258        atom.trust_pqc = trust::RAW as u32;
259        Ok(())
260    }
261
262    fn add_noise(&self, atom: &mut SemanticAtom) -> Result<()> {
263        // Add controlled noise to value
264        let noise = (self.hash_u32(atom.value_fixed) % 100) as i32 - 50; // ±50 units
265        let new_value = (atom.value_fixed as i32 + noise).max(0) as u32;
266        atom.value_fixed = new_value;
267        Ok(())
268    }
269
270    fn hash_mask_value(&self, atom: &mut SemanticAtom) -> Result<()> {
271        // Hash-based value masking
272        atom.value_fixed = self.hash_u32(atom.value_fixed);
273        Ok(())
274    }
275
276    fn apply_differential_privacy(&self, atom: &mut SemanticAtom) -> Result<()> {
277        // Simple differential privacy - Laplace noise
278        // In production, use proper DP library
279        let scale = 1.0 / self.config.privacy_budget;
280        let noise = self.laplace_noise(scale);
281        let noisy_value = atom.get_value() + noise;
282        atom.value_fixed = (noisy_value * crate::FIXED_POINT_PRECISION as f64) as u32;
283        Ok(())
284    }
285
286    fn bucketize_value(&self, atom: &mut SemanticAtom) -> Result<()> {
287        // Round to nearest bucket of 10 units
288        const BUCKET_SIZE: u32 = 1000; // 10.00 in hundredths
289        atom.value_fixed = (atom.value_fixed / BUCKET_SIZE) * BUCKET_SIZE;
290        Ok(())
291    }
292
293    // Utility methods
294
295    fn hash_u32(&self, value: u32) -> u32 {
296        // Simple hash function - in production use proper cryptographic hash
297        let mut hash = value.wrapping_mul(0x9e3779b9);
298        hash ^= hash >> 16;
299        hash = hash.wrapping_mul(0x85ebca6b);
300        hash ^= hash >> 13;
301        hash = hash.wrapping_mul(0xc2b2ae35);
302        hash
303    }
304
305    fn hash_u16(&self, value: u16) -> u16 {
306        let hash = self.hash_u32(value as u32);
307        (hash >> 16) as u16
308    }
309
310    fn laplace_noise(&self, scale: f64) -> f64 {
311        // Simple Laplace noise generator
312        // In production, use proper statistical library
313        let uniform = (self.hash_u32(scale as u32) % 1000) as f64 / 1000.0;
314        let sign = if uniform < 0.5 { -1.0 } else { 1.0 };
315        sign * scale * self.log_approximation(-uniform + 1.0)
316    }
317
318    /// Simple natural logarithm approximation for no_std
319    fn log_approximation(&self, x: f64) -> f64 {
320        if x <= 0.0 {
321            return 0.0;
322        }
323        // Simple approximation: ln(x) ≈ 2 * (x - 1) / (x + 1) for x > 0
324        // This is a rough approximation, good enough for noise generation
325        let y = (x - 1.0) / (x + 1.0);
326        2.0 * y
327    }

343        let shield = Shield::new();
344        let mut atom = AtomBuilder::new()
345            .entity_id(12345)
346            .node_id(678)
347            .value(42.5)
348            .build();
349
